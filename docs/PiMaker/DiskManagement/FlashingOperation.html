    # File lib/pi_maker/disk_management/flashing_operation.rb, line 26
def call
  raise Errno::ENOENT 'Image does not exist' unless File.exist?(image_path)

  @pipe = IO.popen(
    "sudo dd bs=1m if=#{image_path} of=#{disk.raw_disk_path}",
    err: %i[child out]
  )

  self
end
