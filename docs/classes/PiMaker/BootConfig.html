def initialize(opts = {})
  @config = OpenStruct.new(
    opts.fetch(
      :config,
      {
        'dtparam=audio' => 'on',
        'dtoverlay' => 'vc4-fkms-v3d',
        'max_framebuffers' => '2'
      }
    )
  )

  @path = opts[:path]

  @path ||= case PiMaker.host_os
            when :mac
              '/Volumes/boot'
            when :linux, :raspberrypi
              '/mnt/boot'
            else
              'E:/'
            end
end

def to_s
  config.to_h.map do |k, v|
    "#{k}=#{v}"
  end.join("\n")
end
