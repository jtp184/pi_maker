# File lib/pi_maker/command.rb, line 33
def editor_text(txt = '')
  require 'tty-editor'

  preferred = ENV['EDITOR']
  present = first_exec(*TTY::Editor::EXECUTABLES.map { |e| e.split(' ').first })

  tf = Tempfile.create
  tf.write(txt)
  tf.rewind

  system("#{preferred || present} #{tf.path}")

  File.read(tf.path)
end

def prompt(options = {})
  require 'tty-prompt'
  TTY::Prompt.new(
    {
      interrupt: proc do
        puts
        puts 'Exiting...'
        abort
      end
    }.merge(options)
  )
end
